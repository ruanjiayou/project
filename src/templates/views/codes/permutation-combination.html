    // 计算阶乘
    function factorial(n){
        let res = 1
        if(typeof n !== 'number' || n<=0){
            return res;
        }
        while(n) res *= n--;
        return res;
    }
    // 计算排列数
    function arrangement(n,m){
        return factorial(n)/factorial(n-m);
    }
    // 计算组合数
    function combination(n,m){
        return arrangement(n,m)/factorial(m);
    }
