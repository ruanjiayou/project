/**
     * 求连续子数组的最大和
     * @param {array} arr 数组
     */
    function maxSubArr(arr) {
        let len = arr.length,
            maxSum = 0,
            sum = 0;
        for (let i = 0; i < len; i++) {
            sum = sum < 0 ? arr[i] : sum + arr[i];
            if (sum > maxSum) {
                maxSum = sum;
            }
        }
        return maxSum;
    }
